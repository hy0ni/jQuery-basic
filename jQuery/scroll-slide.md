```javacsript
$(document).ready(function(){// $(function(){});축약형

  //gnb리스트 클릭 시 train 이동
  $(".gnb > li").click(function(){

    var idx = $(this).index();
    $(".train").css("left",-1000*idx+"px");
    // Artwork01 클릭 ->left:0
    // Artwork06 클릭 ->left:-1000px
    // Artwork11 클릭 ->left:-2000px
    // Artwork16 클릭 ->left:-3000px

  });

  var count = 0;

  $(".train").on("mousewheel",function(e){
      e.preventDefault();
      var E = e.originalEvent;
      delta = 0;
      if (E.detail == true) {
        //브라우저가 파이어폭스인 경우
          delta = E.detail * -40;
      }else{
        //브라우저가 크롬 익스 오페라인 경우
          delta = E.wheelDelta;
      };

      if(delta>0){
        // 마우스 췰을 올린경우 동작 입력
        count--;
        if(count < 0){
          count = 0;
        }
        $(".train").css("left",-count*200+"px");
      }else{
        // 마우스 휠을 내린경우 동작 입력
        count++;
        if(count > 15){
          count = 15;
        }
        $(".train").css("left",-count*200+"px");
      }
    });

    //gnd안의 리스트를 클릭했을 때 활성화
    $(".train > li").click(function(){
      $(".train > li").removeClass("on");
      $(this).addClass("on");

      // 큰 이미지
    var imgSrc = $(this).find("a").attr("href");
    $(this).children(".bigPic").children("img").attr("src",imgSrc);

    //클릭할때마다 조금씩 오른쪽으로 가기
    var trainIdx = $(this).index();
    if(trainIdx > 15){
      trainIdx = 15;
    }
    $(".train").css("left",-trainIdx*200+"px");
    });

    // 닫기버튼 눌렀을 때
    $(".btnClose").click(function(){
      $(".train > li").removeClass("on");
      return false; // 클릭 이벤트가 상위요소로 전달되는 것을 억제
    });
});
```
